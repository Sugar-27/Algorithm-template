### 快速排序

```c++
void quickSort(vector<int>& nums, int l, int r) {
    if (l >= r)
        return;
    int x = nums[(l + r) >> 1];
    int i = l, j = r;
    while (i <= j) {
        while (nums[i] < x)
            ++i;
        while (nums[j] > x)
            --j;
        if (i == j)
            break;
        if (i < j)
            swap(nums[i++], nums[j--]);
    }
    quickSort(nums, l, j);
    quickSort(nums, j + 1, r);
}
```
